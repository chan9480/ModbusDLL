cmake_minimum_required(VERSION 3.15)

# Use C:/vcpkg toolchain when not specified on the command line
if(NOT DEFINED CMAKE_TOOLCHAIN_FILE)
    set(CMAKE_TOOLCHAIN_FILE
        "C:/vcpkg/scripts/buildsystems/vcpkg.cmake"
        CACHE STRING "Vcpkg toolchain file" FORCE)
endif()

set(VCPKG_TARGET_TRIPLET "x64-windows" CACHE STRING "vcpkg target triplet" FORCE)

project(ModbusServer LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(libmodbus CONFIG REQUIRED)

add_library(ModbusServer SHARED ModbusServer.cpp)

set_target_properties(
    ModbusServer
    PROPERTIES
    OUTPUT_NAME "ModbusServer"
)

target_link_libraries(ModbusServer PRIVATE modbus)

if(MSVC)
    target_compile_options(ModbusServer PRIVATE /W4 /EHsc)
endif()

# Copy modbus-5.dll next to ModbusServer.dll so the Windows loader can find it
add_custom_command(TARGET ModbusServer POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "C:/vcpkg/installed/x64-windows/bin/modbus-5.dll"
        "$<TARGET_FILE_DIR:ModbusServer>"
    COMMENT "Copying modbus-5.dll to ModbusServer output directory"
)

# ── ModbusClient DLL ─────────────────────────────────────────────────────────

add_library(ModbusClient SHARED ModbusClient.cpp)

set_target_properties(
    ModbusClient
    PROPERTIES
    OUTPUT_NAME "ModbusClient"
)

target_link_libraries(ModbusClient PRIVATE modbus)

if(MSVC)
    target_compile_options(ModbusClient PRIVATE /W4 /EHsc)
endif()

# Copy modbus-5.dll next to ModbusClient.dll as well
add_custom_command(TARGET ModbusClient POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "C:/vcpkg/installed/x64-windows/bin/modbus-5.dll"
        "$<TARGET_FILE_DIR:ModbusClient>"
    COMMENT "Copying modbus-5.dll to ModbusClient output directory"
)
